5#[derive(Clone, Debug, Eq, PartialEq, Hash)]
6pub struct Metadata {

8    pub year: Option<i64>,
9    pub season: Option<i64>,
10    pub episode: Option<i64>,
11}

13pub trait FilenameMetadata {
14    fn from_str(s: &str) -> Option<Metadata>;
15}

17impl FilenameMetadata for TorrentMetadata {
18    fn from_str(s: &str) -> Option<Metadata> {
19        let metadata = TorrentMetadata::from(s).ok()?;
20
21        Some(Metadata {
22            name: metadata.title().to_owned(),
23            year: metadata.year().map(|x| x as i64),
24            season: metadata.season().map(|x| x as i64),
25            episode: metadata.episode().map(|x| x as i64),
26        })
27    }
28}

30impl FilenameMetadata for Anitomy {
31    fn from_str(s: &str) -> Option<Metadata> {
32        let metadata = match Anitomy::new().parse(s) {
33            Ok(v) | Err(v) => v,
34        };
35
36        Some(Metadata {
37            name: metadata.get(ElementCategory::AnimeTitle)?.to_string(),
38            year: metadata
39                .get(ElementCategory::AnimeYear)
40                .and_then(|x| x.parse().ok()),
41            // If season isnt specified we assume season 1 here.
42            season: metadata
43                .get(ElementCategory::AnimeSeason)
44                .and_then(|x| x.parse().ok())
45                .or(Some(1)),
46            episode: metadata
47                .get(ElementCategory::EpisodeNumber)
48                .and_then(|x| x.parse().ok()),
49        })
50    }
51}

53/// A special filename metadata extractor that combines torrent_name_parser and anitomy which in
54/// some cases is necessary. TNP is really good at extracting show titles but not season and
55/// episode numbers. Anitomy excels at this. Here we combine the title extracted by TPN and the
56/// season and episode number extracted by Anitomy.
57pub struct CombinedExtractor;
58
59impl FilenameMetadata for CombinedExtractor {
60    fn from_str(s: &str) -> Option<Metadata> {
61        let metadata_tnp = TorrentMetadata::from(s).ok()?;
62        let metadata_anitomy = match Anitomy::new().parse(s) {
63            Ok(v) | Err(v) => v,
64        };
65
66        Some(Metadata {
67            name: metadata_tnp.title().to_owned(),
68            year: metadata_tnp.year().map(|x| x as i64),
69            // If season isnt specified we assume season 1 here as some releases only have a
70            // episode number and no season number.
71            season: metadata_anitomy
72                .get(ElementCategory::AnimeSeason)
73                .and_then(|x| x.parse().ok())
74                .or(Some(1)),
75            episode: metadata_anitomy
76                .get(ElementCategory::EpisodeNumber)
77                .and_then(|x| x.parse().ok()),
78        })
79    }
