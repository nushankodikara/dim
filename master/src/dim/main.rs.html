1use std::path::PathBuf;
2use std::time::Duration;
3
4use clap::Parser;
5use dim::streaming;
6
7use xtra::spawn::Tokio;
8
9use dim_core as dim;
10#[derive(Debug, clap::Parser)]
11#[clap(name = "Dim", about = "Dim, a media manager fueled by dark forces.")]
12#[clap(version = env!("CARGO_PKG_VERSION"), author = env!("CARGO_PKG_AUTHORS"))]
13#[clap(rename_all = "kebab")]
14struct Args {
15    #[clap(short, long, env = "DIM_CONFIG_PATH")]
16    config: Option<PathBuf>,
17}
18
19fn main() {
20    let args = Args::parse();
21    let _ = std::fs::create_dir_all(dim::utils::ffpath("config"));
22
23    let config_path = args
24        .config
25        .map(|x| x.to_string_lossy().to_string())
26        .unwrap_or(dim::utils::ffpath("config/config.toml"));
27
28    // initialize global settings.
29    dim::init_global_settings(Some(config_path)).expect("Failed to initialize global settings.");
30
31    let global_settings = dim::get_global_settings();
32
33    // never panics because we set a default value to metadata_dir
34    let _ = std::fs::create_dir_all(global_settings.metadata_dir.clone());
35
36    // set our jwt secret key
37    let settings_clone = global_settings.clone();
38    let secret_key = global_settings.secret_key.unwrap_or_else(move || {
39        let secret_key = dim_database::generate_key();
40        dim::set_global_settings(dim::GlobalSettings {
41            secret_key: Some(secret_key),
42            ..settings_clone
43        })
44        .expect("Failed to save JWT secret_key.");
45        secret_key
46    });
47
48    dim_database::set_key(secret_key);
49
50    dim_core::core::METADATA_PATH
51        .set(global_settings.metadata_dir.clone())
52        .expect("Failed to set METADATA_PATH");
53
54    dim::setup_logging(global_settings.verbose);
55
56    {
57        let failed = streaming::ffcheck()
58            .into_iter()
59            .fold(false, |failed, item| match item {
60                Ok(stdout) => {
61                    tracing::info!("{}", stdout);
62                    failed
63                }
64
65                Err(program) => {
66                    tracing::error!("Could not find: {}", program);
67                    true
68                }
69            });
70
71        if failed {
72            // FIXME: I think in some cases we exit so fast that the error above is not printed out
73            // or just partially printed out.
74            std::process::exit(1);
75        }
76    }
77
78    // The mediafile scanner is super hungry for fds. Increase our limits here as much as possible.
79    if let Some(limit) = fdlimit::raise_fd_limit() {
80        tracing::info!(limit, "Raising fd limit.");
81    }
82
83    nightfall::profiles::profiles_init(crate::streaming::FFMPEG_BIN.to_string());
84
85    let async_main = async move {
86        let (event_tx, event_rx) = tokio::sync::mpsc::unbounded_channel();
87        let pool = dim_database::get_conn().await.unwrap();
88
89        // Before we start making DB-calls we need to initialize our CDC pipeline.
90        {
91            let mut lock = pool.writer().lock_owned().await;
92            let mut reactor_core = dim::reactor::ReactorCore::new();
93            reactor_core.register(&mut lock).await;
94
95            let reactor = dim::reactor::handler::EventReactor::new(pool.clone())
96                .with_websocket(event_tx.clone());
97
98            tokio::spawn(reactor_core.react(reactor));
99        }
100
101        let stream_manager = nightfall::StateManager::new(
102            &mut Tokio::Global,
103            global_settings.cache_dir.clone(),
104            crate::streaming::FFMPEG_BIN.to_string(),
105        );
106
107        let stream_manager_clone = stream_manager.clone();
108
109        // GC the stream manager every 100ms
110        tokio::spawn(async move {
111            let mut interval = tokio::time::interval(Duration::from_millis(1000));
112            interval.tick().await;
113
114            loop {
115                interval.tick().await;
116                let _ = stream_manager_clone.garbage_collect().await.unwrap();
117            }
118        });
119
120        if !global_settings.quiet_boot {
121            tracing::info!("Scanning for media files...");
122            dim::core::run_scanners(event_tx.clone()).await;
123        }
124
125        tracing::info!("Launcing Dim");
126
127        let address = std::net::SocketAddr::new(
128            std::net::IpAddr::V4(std::net::Ipv4Addr::new(0, 0, 0, 0)),
129            global_settings.port,
130        );
131
132        dim_web::start_webserver(address, event_tx, stream_manager, event_rx, async move {
133            let _ = tokio::signal::ctrl_c().await;
134            tracing::info!("CTRL-C received, shutting down...");
135        })
136        .await;
137    };
138
139    tokio::runtime::Runtime::new()
140        .expect("Failed to create a tokio runtime.")
141        .block_on(async_main);
142}
