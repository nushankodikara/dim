31/// This semaphore is necessary so that we dont create too many instances of `MediafileCreator`.
32/// Having too many instances would not make sense as we can support only so many instances without
33/// ending up contending over the database lock.
34static SEMPAHORE: Semaphore = Semaphore::const_new(12);
35
36pub type Result<T> = ::core::result::Result<T, Error>;
37pub type SqlxError = Arc<sqlx::Error>;
38
39#[derive(Clone, Debug, Display, Error, Serialize)]
40pub enum Error {
41    /// The file already exists in the database.
42    FileExists,
43    /// Failed to acquire a read-only database transaction: {0:?}
44    FailedToAcquireReader(#[serde(skip)] SqlxError),
45    /// Failed to acquire a read-write database transaction: {0:?}
46    FailedToAcquireWriter(#[serde(skip)] SqlxError),
47    /// File passed is non-unicode.
48    NonUnicodeFile,
49    /// Failed to extract media information with ffprobe: {0:?}
50    FfprobeError(#[serde(skip)] Arc<std::io::Error>),
51    /// Failed to write mediafile to the database: {0:?}
52    InsertFailed(#[serde(skip)] DatabaseError),
53    /// Failed to select written mediafile from the database: {0:?}
54    SelectFailed(#[serde(skip)] DatabaseError),
55    /// Failed to commit mediafiles to the database: {0:?}
56    CommitFailed(#[serde(skip)] SqlxError),
57    /// Failed to check if mediafile exists in the database: {0:?}
58    ExistanceCheckFailed(#[serde(skip)] DatabaseError),
59}

61/// Struct is responsible for creating `InsertableMediaFile`'s and preparing them for insertions
62/// into the database. It also handles writing these files to the database, and is responsible for
63/// various optimizations such as batching.
64///
65/// The number of instances of this struct that can exist at any given time is restricted by a
66/// semaphore. If this limit is reached, `MediafileCreator::new` will wait until some slots
67/// free up. This is done as an optimization so that we don't have so many instances of this
68/// that we start contending over the database lock, inducing timeouts and performance bottlenecks.
69pub struct MediafileCreator {
70    /// Pool of database connections.
71    conn: DbConnection,
72    /// Library which we will assign as the owner of the mediafiles inserted by this instance.
73    library_id: i64,
74    /// Represents a permit that we must own for the lifetime of the instance of `Self`. It is
75    /// important that we use permits so that we can control how many instances of `Self` exist at
76    /// any point in time.
77    _permit: SemaphorePermit<'static>,
78}
79
80impl MediafileCreator {
81    /// Create a new instance of `MediafileCreator`. A instance will be returned once a slot in our
82    /// sempahore frees up.
83    pub async fn new(conn: DbConnection, library_id: i64) -> Self {
84        // NOTE: This will never panic because `SEMPAHORE` will never be dropped.
85        let permit = SEMPAHORE
86            .acquire()
87            .await
88            .expect("Failed to acquire a permit for `MediafileCreator`");
89
90        Self {
91            conn,
92            library_id,
93            _permit: permit,
94        }
95    }
96
97    /// Method constructs a `InsertableMediaFile` from a path to a file and the metadata extracted
98    /// from its filename. In addition to using metadata from the filename, it also spawns ffprobe
99    /// to obtain stream information. This method can be called concurrently.
100    ///
101    /// # Return
102    /// In addition to normal errors, this method can return `Error::FileExists`. In this case,
103    /// this method has identified that a file with the path passed already exists in the database.
104    /// This isnt a hard error and is simply meant to reduce database load. This error can just be
105    /// skipped.
106    ///
107    /// # Database access
108    /// This method creates one read-only database transaction and it should not affect the
109    /// concurrency of read-write transactions.
110    #[tracing::instrument(skip(self))]
111    pub async fn construct_mediafile(
112        &self,
113        file: PathBuf,
114        metadata: Metadata,
115    ) -> Result<InsertableMediaFile> {
116        let target_file = file.to_str().ok_or(Error::NonUnicodeFile)?.to_owned();
117
118        // NOTE: This is intended to be a fast-path in-case we have inserted the file into the
119        // database a long time ago. Downstream consumers of `InsertableMediaFile` should still
120        // double-check that the file isnt in the database already.
121        {
122            let mut tx = self
123                .conn
124                .read()
125                .begin()
126                .await
127                .map_err(|e| Error::FailedToAcquireReader(e.into()))?;
128
129            if MediaFile::exists_by_file(&mut tx, &target_file).await {
130                return Err(Error::FileExists);
131            }
132        }
133
134        // FIXME: This is a huge bottleneck. FFProbe is slow. I'm guessing one of the reasons is
135        // that it is not embedded but rather we access an on-disk binary, and the second reason is
136        // that we are io-bound here.
137        //
138        // To fix this we could have a actor that populates metadata information at its own pace
139        // and does its scheduling. If a user ever needs to obtain the metadata, we can request it
140        // and patch it immediately. This would add a initial cost to the API call, but subsequent
141        // API calls will be cheap.
142        let video_metadata = match FFProbeCtx::new(&FFPROBE_BIN).get_meta(&target_file).await {
143            Ok(x) => x,
144            Err(error) => {
145                error!(?error, "Couldn't extract media information with ffprobe");
146                error!(file = &target_file, "Assuming file is corrupted.");
147                Default::default()
148            }
149        };
150
151        Ok(InsertableMediaFile {
152            library_id: self.library_id,
153            media_id: None,
154            target_file,
155
156            raw_name: metadata.name,
157            raw_year: metadata.year,
158            season: metadata.season,
159            episode: metadata.episode,
160
161            quality: video_metadata.get_height().map(|x| x.to_string()),
162            codec: video_metadata.get_video_codec(),
163            container: Some(video_metadata.get_container()),
164            audio: video_metadata
165                .get_primary_codec("audio")
166                .map(ToOwned::to_owned),
167            original_resolution: Default::default(),
168            duration: video_metadata.get_duration().map(|x| x as i64),
169            corrupt: Some(video_metadata.is_corrupt()),
170            channels: video_metadata.get_primary_channels(),
171            profile: video_metadata.get_video_profile(),
172            audio_language: video_metadata
173                .get_audio_lang()
174                .or_else(|| video_metadata.get_video_lang())
175                .as_deref()
176                .and_then(crate::utils::lang_from_iso639)
177                .map(ToString::to_string),
178        })
179    }
180
181    /// Method will insert a batch of `InsertableMediaFile` within the context of one transaction.
182    /// Before inserting a file it will attempt to look up if it is already in the database, and if
183    /// so it will skip it.
184    ///
185    /// For better parallelism, this method can be called through the `Actor` interface.
186    ///
187    /// # Return
188    /// Method will return a list of mediafiles that should now be in the database.
189    ///
190    /// # Tracing spans
191    /// Method instruments several futures:
192    ///
193    /// * `mediafile_insert` - tracks when a insert operation is called.
194    /// * `mediafile_select` - tracks the select for a mediafile that happens after a insert.
195    /// * `database_commit` - tracks when a database commit happens.
196    #[tracing::instrument(skip(self, batch))]
197    pub async fn insert_batch<'a>(
198        &mut self,
199        batch: impl Iterator<Item = &'a InsertableMediaFile>,
200    ) -> Result<Vec<MediaFile>> {
201        let mut work_done = vec![];
202
203        let mut lock = self.conn.writer().lock_owned().await;
204        let mut tx = dim_database::write_tx(&mut lock)
205            .await
206            .map_err(|e| Error::FailedToAcquireWriter(e.into()))?;
207
208        for mediafile in batch {
209            if mediafile
210                .exists(&mut tx)
211                .await
212                .map_err(Error::ExistanceCheckFailed)?
213            {
214                warn!(?mediafile, "Mediafile already exists in the database.");
215                continue;
216            }
217
218            let id = mediafile
219                .insert(&mut tx)
220                .instrument(debug_span!("mediafile_insert"))
221                .await
222                .map_err(Error::InsertFailed)?;
223
224            work_done.push(
225                MediaFile::get_one(&mut tx, id)
226                    .instrument(debug_span!("mediafile_select"))
227                    .await
228                    .map_err(Error::SelectFailed)?,
229            );
230        }
231
232        tx.commit()
233            .instrument(debug_span!("database_commit"))
234            .await
235            .map_err(|e| Error::CommitFailed(e.into()))?;
236
237        Ok(work_done)
238    }
239}
