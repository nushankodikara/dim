44#[derive(Clone, Debug, Display, Error, Serialize)]
45pub enum Error {
46    /// Failed to insert poster into database: {0:?}
47    PosterInsert(#[serde(skip)] dim_database::DatabaseError),
48    /// Failed to insert backdrop into database: {0:?}
49    BackdropInsert(#[serde(skip)] dim_database::DatabaseError),
50    /// Failed to decouple genres from media: {0:?}
51    GenreDecouple(#[serde(skip)] dim_database::DatabaseError),
52    /// Failed to create or get genre: {0:?}
53    GetOrInsertGenre(#[serde(skip)] dim_database::DatabaseError),
54    /// Failed to attach genre to media object: {0:?}
55    CoupleGenre(#[serde(skip)] dim_database::DatabaseError),
56    /// Failed to update mediafile to point to new parent: {0:?}
57    UpdateMediafile(#[serde(skip)] dim_database::DatabaseError),
58    /// Failed to get children count for movie: {0:?}
59    ChildrenCount(#[serde(skip)] dim_database::DatabaseError),
60    /// Failed to cleanup child-less parent: {0:?}
61    ChildCleanup(#[serde(skip)] dim_database::DatabaseError),
62    /// Failed to insert or get tv object: {0:?}
63    GetOrInsertMedia(#[serde(skip)] dim_database::DatabaseError),
64    /// Failed to insert or get season: {0:?}
65    GetOrInsertSeason(#[serde(skip)] dim_database::DatabaseError),
66    /// Failed to insert media object for episode: {0:?}
67    GetOrInsertMediaEpisode(#[serde(skip)] dim_database::DatabaseError),
68    /// Failed to insert episode object: {0:?}
69    GetOrInsertEpisode(#[serde(skip)] dim_database::DatabaseError),
70    /// Failed to get season id for episode: {0:?}
71    GetSeasonId(#[serde(skip)] dim_database::DatabaseError),
72    /// Failed to get tvshowid for season: {0:?}
73    GetTvId(#[serde(skip)] dim_database::DatabaseError),
74    /// Season not found
75    SeasonNotFound,
76    /// Episode not found
77    EpisodeNotFound,
78}

84    async fn match_to_result<'life0>(
85        &self,
86        tx: &mut Transaction<'life0>,
87        file: MediaFile,
88        result: (ExternalMedia, ExternalSeason, ExternalEpisode),
89    ) -> Result<(i64, i64, i64), Error> {
90        // TODO: insert poster and backdrops.
91        let (emedia, eseason, eepisode) = result;
92
93        let posters = emedia
94            .posters
95            .iter()
96            .filter_map(|x| asset_from_url(x))
97            .collect::<Vec<_>>();
98
99        let mut poster_ids = vec![];
100
101        for poster in posters {
102            let asset = poster
103                .insert(&mut *tx)
104                .await
105                .inspect_err(|error| error!(?error, "Failed to insert asset into db."))
106                .map_err(Error::PosterInsert)?;
107
108            poster_ids.push(asset);
109        }
110
111        let backdrops = emedia
112            .backdrops
113            .iter()
114            .filter_map(|x| asset_from_url(x))
115            .collect::<Vec<_>>();
116
117        let mut backdrop_ids = vec![];
118
119        for backdrop in backdrops {
120            let asset = backdrop
121                .insert(&mut *tx)
122                .await
123                .inspect_err(|error| error!(?error, "Failed to insert asset into db."))
124                .map_err(Error::BackdropInsert)?;
125
126            backdrop_ids.push(asset);
127        }
128
129        let media = InsertableMedia {
130            media_type: MediaType::Tv,
131            library_id: file.library_id,
132            name: emedia.title,
133            description: emedia.description,
134            rating: emedia.rating,
135            year: emedia.release_date.map(|x| x.year() as _),
136            added: Utc::now().to_string(),
137            poster: poster_ids.first().map(|x| x.id),
138            backdrop: backdrop_ids.first().map(|x| x.id),
139        };
140
141        let parent_id = media
142            .lazy_insert(tx)
143            .await
144            .inspect_err(|error| error!(?error, ?file, "Failed to lazy insert tv show"))
145            .map_err(Error::GetOrInsertMedia)?;
146
147        // NOTE: We want to decouple this media from all genres and essentially rebuild the list.
148        // Its a lot simpler than doing a diff-update but it might be more expensive.
149        Genre::decouple_all(tx, parent_id)
150            .await
151            .inspect_err(|error| error!(?error, "Failed to decouple genres from media."))
152            .map_err(Error::GenreDecouple)?;
153
154        for name in emedia.genres {
155            let genre = InsertableGenre { name }
156                .insert(tx)
157                .await
158                .inspect_err(|error| error!(?error, "Failed to create or get genre."))
159                .map_err(Error::GetOrInsertGenre)?;
160
161            // TODO: Recouple genres always otherwise rematching would get buggy genre lists
162            InsertableGenreMedia::insert_pair(genre, parent_id, tx)
163                .await
164                .inspect_err(
165                    |error| error!(?error, %parent_id, "Failed to attach genre to media object."),
166                )
167                .map_err(Error::CoupleGenre)?;
168        }
169
170        let seasonid = self.match_to_season(tx, parent_id, eseason).await?;
171        let episodeid = self
172            .match_to_episode(tx, file.clone(), seasonid, eepisode)
173            .await?;
174
175        // If the mediafile used to belong to a different episode/season/show we want to
176        // recursively search if we need to delete the parents. If the parents have 0 children, we
177        // want to erase their existance.
178        match file.media_id {
179            Some(x) if x != episodeid => {
180                let season_id = Episode::get_seasonid(tx, x)
181                    .await
182                    .inspect_err(
183                        |error| error!(?error, id = %x, "Failed to get seasonid for episode"),
184                    )
185                    .map_err(Error::GetSeasonId)?;
186
187                let tvshow_id = Season::get_tvshowid(tx, season_id).await.inspect_err(
188                    |error| error!(?error, id = %x, "Failed to get tvshowid for season/episode."),
189                ).map_err(Error::GetTvId)?;
190
191                let count = Movie::count_children(tx, x).await.inspect_err(
192                    |error| error!(?error, id = %x, "Failed to obtain children count for episode."),
193                ).map_err(Error::ChildrenCount)?;
194
195                if count == 0 {
196                    Media::delete(tx, x)
197                        .await
198                        .inspect_err(
199                            |error| error!(?error, id = %x, "Failed to delete child-less episode"),
200                        )
201                        .map_err(Error::ChildCleanup)?;
202                }
203
204                let count = Season::count_children(tx, season_id)
205                    .await
206                    .inspect_err(
207                        |error| error!(?error, id = %x, "Failed to get children count for season"),
208                    )
209                    .map_err(Error::ChildrenCount)?;
210
211                if count == 0 {
212                    Season::delete_by_id(tx, season_id)
213                        .await
214                        .inspect_err(
215                            |error| error!(?error, id = %x, "Failed to delete child-less season"),
216                        )
217                        .map_err(Error::ChildCleanup)?;
218                }
219
220                let count = TVShow::count_children(tx, tvshow_id).await.inspect_err(
221                    |error| error!(?error, id = %x, "Failed to get children count for tv show."),
222                ).map_err(Error::ChildrenCount)?;
223
224                if count == 0 {
225                    Media::delete(tx, tvshow_id)
226                        .await
227                        .inspect_err(
228                            |error| error!(?error, id = %x, "Failed to delete child-less tv show"),
229                        )
230                        .map_err(Error::ChildCleanup)?;
231                }
232            }
233            _ => {}
234        }
235
236        Ok((parent_id, seasonid, episodeid))
237    }

241    async fn match_to_season(
242        &self,
243        tx: &mut Transaction<'_>,

253        let mut poster_ids = vec![];
254

256            let asset = poster

258                .await

280    async fn match_to_episode(
281        &self,
282        tx: &mut Transaction<'_>,
283        file: MediaFile,
284        seasonid: i64,
285        result: ExternalEpisode,
286    ) -> Result<i64, Error> {
287        let stills = result
288            .stills
289            .iter()
290            .filter_map(|x| asset_from_url(x))
291            .collect::<Vec<_>>();
292
293        let mut still_ids = vec![];
294
295        for still in stills {
296            let asset = still
297                .insert(&mut *tx)
298                .await
299                .inspect_err(|error| error!(?error, "Failed to insert asset into db."))
300                .map_err(Error::PosterInsert)?;
301
302            still_ids.push(asset);
303        }
304
305        let media = InsertableMedia {
306            library_id: file.library_id,
307            name: result.title_or_episode(),
308            added: Utc::now().to_string(),
309            media_type: MediaType::Episode,
310            description: result.description.clone(),
311            backdrop: still_ids.first().map(|x| x.id),
312            ..Default::default()
313        };
314
315        let episode = InsertableEpisode {
316            episode: result.episode_number as _,
317            seasonid,
318            media,
319        };
320
321        episode
322            .media
323            .insert_blind(&mut *tx)
324            .await
325            .inspect_err(|error| error!(?error, ?file, "Failed to insert media for episode."))
326            .map_err(Error::GetOrInsertMediaEpisode)?;
327
328        // NOTE: WE use to turn a episode into a movie here, not sure if necessary anymore.
329
330        let episode_id = episode
331            .insert(&mut *tx)
332            .await
333            .inspect_err(|error| error!(?error, ?file, "Failed to insert episode."))
334            .map_err(Error::GetOrInsertEpisode)?;
335
336        let updated_mediafile = UpdateMediaFile {
337            media_id: Some(episode_id),
338            ..Default::default()
339        };
340
341        updated_mediafile
342            .update(&mut *tx, file.id)
343            .await
344            .inspect_err(|error| error!(?error, ?file, "Failed to update mediafile media id."))
345            .map_err(Error::UpdateMediafile)?;
346
347        Ok(episode_id)
348    }

350    #[instrument(skip(provider, metadata))]
351    async fn lookup_metadata(
352        provider: Arc<dyn ExternalQueryShow>,
353        file: MediaFile,
354        metadata: Vec<Metadata>,
355    ) -> Option<(MediaFile, (ExternalMedia, ExternalSeason, ExternalEpisode))> {
356        for meta in metadata {
357            match provider
358                .search(meta.name.as_ref(), meta.year.map(|x| x as _))
359                .await
360            {
361                Ok(provided) => {
362                    let first = if let Some(x) = provided.first() {
363                        x.clone()
364                    } else {
365                        continue;
366                    };
367
368                    let Ok(seasons) = provider.seasons_for_id(&first.external_id).await else {
369                        info!(?meta, "Failed to find season match with the current metadata set.");
370                        continue;
371                    };
372
373                    // FIXME: If a file doesnt have season metadata, we want to default to
374                    // marking this file as an extra and put it in season 0
375                    let Some(season) = seasons
376                        .into_iter()
377                        .find(|x| x.season_number as i64 == meta.season.unwrap_or(0)) else {
378                        info!(?meta, "Provider didnt return our desired season with current metadata.");
379                        continue;
380                    };
381
382                    let Ok(episodes) = provider
383                        .episodes_for_season(&first.external_id, meta.season.unwrap_or(0) as _)
384                        .await else {
385                        // FIXME: We might want to propagate this error.
386                        info!(?meta, "Failed to fetch episodes with current metadata set.");
387                        continue;
388                    };
389
390                    let Some(episode) = episodes
391                        .into_iter()
392                        .find(|x| x.episode_number as i64 == meta.episode.unwrap_or(0)) else {
393                        info!(
394                            ?meta,
395                            "Provider didnt return our desired episode with current metadata."
396                        );
397                        continue;
398                    };
399
400                    return Some((file, (first, season, episode)));
401                }
402                Err(e) => error!(?meta, error = ?e, "Failed to find a movie match."),
403            }
404        }
405
406        None
407    }

411impl MediaMatcher for TvMatcher {
412    async fn batch_match(
413        &self,
414        tx: &mut Transaction<'_>,
415        provider: Arc<dyn ExternalQueryIntoShow>,
416        work: Vec<WorkUnit>,
417    ) -> Result<(), super::Error> {
418        let provider_show: Arc<dyn ExternalQueryShow> = provider
419            .into_query_show()
420            .expect("Scanner needs a show provider");
421
422        let metadata_futs = work
423            .into_iter()
424            .map(|WorkUnit(file, metadata)| {
425                let provider_show = Arc::clone(&provider_show);
426                tokio::spawn(Self::lookup_metadata(provider_show, file, metadata))
427            })
428            .collect::<Vec<_>>();
429
430        let metadata = futures::future::join_all(metadata_futs).await;
431
432        for meta in metadata.into_iter() {
433            if let Ok(Some((file, provided))) = meta {
434                self.match_to_result(tx, file, provided)
435                    .await
436                    .inspect_err(|error| error!(?error, "failed to match to result"))?;
437            }
438        }
439
440        Ok(())
441    }

443    async fn match_to_id(
444        &self,
445        tx: &mut Transaction<'_>,
446        provider: Arc<dyn ExternalQueryIntoShow>,
447        work: WorkUnit,
448        external_id: &str,
449    ) -> Result<(), super::Error> {
450        let provider: Arc<dyn ExternalQueryShow> = provider
451            .into_query_show()
452            .expect("Scanner needs a show provider");
453
454        let WorkUnit(file, metadata) = work;
455
456        let provided = match provider.search_by_id(external_id).await {
457            Ok(provided) => provided,
458            Err(e) => {
459                error!(%external_id, error = ?e, "Failed to find a movie match.");
460                return Err(super::Error::InvalidExternalId);
461            }
462        };
463
464        let mut season_result = None;
465        let mut episode_result = None;
466
467        for meta in metadata {
468            let Ok(seasons) = provider.seasons_for_id(external_id).await else {
469                info!(?meta, "Failed to find season match with the current metadata set.");
470                continue;
471            };
472
473            // FIXME: If a file doesnt have season metadata, we want to default to
474            // marking this file as an extra and put it in season 0
475            let Some(season) = seasons
476                .into_iter()
477                .find(|x| x.season_number as i64 == meta.season.unwrap_or(0)) else {
478                    info!(?meta, "Provider didnt return our desired season with current metadata.");
479                    continue;
480                };
481
482            let Ok(episodes) = provider
483                .episodes_for_season(external_id, meta.season.unwrap_or(0) as _)
484                .await else {
485                    // FIXME: We might want to propagate this error.
486                    info!(?meta, "Failed to fetch episodes with current metadata set.");
487                    continue;
488                };
489
490            let Some(episode) = episodes
491                .into_iter()
492                .find(|x| x.episode_number as i64 == meta.episode.unwrap_or(0)) else {
493                    info!(
494                        ?meta,
495                        "Provider didnt return our desired episode with current metadata."
496                    );
497                    continue
498                };
499
500            season_result = Some(season);
501            episode_result = Some(episode);
502        }
503
504        let Some(season_result) = season_result else { return Err(Error::SeasonNotFound.into()); };
505        let Some(episode_result) = episode_result else { return Err(Error::EpisodeNotFound.into()); };
506
507        self.match_to_result(tx, file, (provided, season_result, episode_result))
508            .await
509            .inspect_err(|error| error!(?error, "failed to match to result"))?;
510
511        Ok(())
512    }
