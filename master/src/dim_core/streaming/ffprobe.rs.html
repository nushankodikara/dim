133impl FFProbeCtx {
134    pub fn new(ffprobe_bin: &'static str) -> Self {
135        Self {
136            ffprobe_bin: ffprobe_bin.to_owned(),
137        }
138    }
139
140    #[tracing::instrument(skip(self, file))]
141    pub async fn get_meta(&self, file: impl ToString) -> Result<FFPStream, std::io::Error> {
142        let mut probe = Command::new(self.ffprobe_bin.clone());
143
144        probe
145            .arg(file.to_string())
146            .arg("-v")
147            .arg("error")
148            .arg("-print_format")
149            .arg("json")
150            .arg("-show_streams")
151            .arg("-show_format")
152            .stdout(Stdio::piped())
153            .stderr(Stdio::piped());
154
155        trace!(
156            binary = self.ffprobe_bin.as_str(),
157            args = %probe.as_std().get_args().filter_map(|x| x.to_str()).collect::<Vec<_>>().join(" "),
158            "Spawning ffprobe."
159        );
160
161        let output = probe.spawn()?.wait_with_output().await?;
162
163        if !output.status.success() {
164            let stderr = String::from_utf8_lossy(output.stderr.as_slice());
165            error!(status = ?output.status, stderr = %stderr, "ffprobe exited with an error status.");
166        }
167
168        let json = String::from_utf8_lossy(output.stdout.as_slice());
169
170        let de = serde_json::from_str(&json).unwrap_or_default();
171
172        Ok(de)
173    }
174}

213    pub fn get_primary(&self, codec_type: &str) -> Option<&Stream> {
214        let mut streams: VecDeque<_> = self.find_by_type(codec_type).into();
215
216        if streams.is_empty() {
217            return None;
218        }
219
220        if streams.len() == 1 {
221            return streams.pop_front();
222        }
223
224        let primary_stream = streams.iter().find_map(|x| {
225            if x.disposition.as_ref()?.default == 1 {
226                Some(*x)
227            } else {
228                None
229            }
230        });
231
232        primary_stream.or_else(|| streams.pop_front())
233    }
234
235    pub fn get_primary_codec(&self, codec_type: &str) -> Option<&str> {
236        Some(&self.get_primary(codec_type)?.codec_name)
237    }
