1//! Dim is a media manager written in rust.
2//! It uses Diesel as the ORM and rocket for the http/s server
3//!
4//! The project is split up into several crates:
5//! * [`database`](dim-database) - Holds all the database models including some frequently used db operations
6//! * [`routes`] - All of the routes that we expose over http are stored in there
7//! * [`scanners`](scanner) - The filesystem scanner and daemon code is located here
8//! ffmpeg that is used by several parts of dim
9//!
10//! # Building
11//! Dim can easily be built with cargo build --release.
12//! When built with --release, build.rs will compile the web ui and embed it into dim.
13//!
14//! # To run
15//! Dim can be ran using docker, by pulling vgarleanu/dim-server, or locally.
16//! If ran locally, make sure PostgreSQL is running with the password for postgres: dimpostgres
17//!
18//! # Testing
19//! To test run `make test` in the root, or `cargo test` in the root of each module including the
20//! root dir.

60/// Function builds a logger drain that drains to a json file located in logs/ and also to stdout.
61pub fn setup_logging(_debug: bool) {
62    let _ = create_dir_all("logs");
63
64    if std::env::var("RUST_LOG").is_err() {
65        std::env::set_var("RUST_LOG", "info,tower_http=trace");
66    }
67
68    let log_appender = tracing_appender::rolling::daily("./logs", "dim-log.log");
69    let (non_blocking_file, _guard) = tracing_appender::non_blocking(log_appender);
70
71    let subscriber = tracing_subscriber::registry()
72        .with(EnvFilter::from_default_env())
73        .with(fmt::layer().with_writer(std::io::stdout))
74        .with(fmt::layer().json().with_writer(non_blocking_file));
75
76    let _ = tracing::subscriber::set_global_default(subscriber);
77}

79#[cfg(test)]
80pub fn setup_test_logging() {
81    if std::env::var("RUST_LOG").is_err() {
82        std::env::set_var("RUST_LOG", "info");
83    }
84
85    let subscriber = tracing_subscriber::registry()
86        .with(EnvFilter::from_default_env())
87        .with(
88            fmt::layer()
89                .with_span_events(fmt::format::FmtSpan::CLOSE | fmt::format::FmtSpan::NEW)
90                .with_writer(tracing_subscriber::fmt::TestWriter::new()),
91        );
92
93    let _ = tracing::subscriber::set_global_default(subscriber);
