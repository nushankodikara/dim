18pub async fn get_user_settings(
19    Extension(user): Extension<User>,
20    State(AppState { conn, .. }): State<AppState>,
21) -> Result<Response, AuthError> {
22    let mut tx = conn.read().begin().await.map_err(DatabaseError::from)?;
23    Ok(axum::response::Json(&User::get_by_id(&mut tx, user.id).await?.prefs).into_response())
24}

26pub async fn post_user_settings(
27    Extension(user): Extension<User>,
28    State(AppState { conn, .. }): State<AppState>,
29    Json(new_settings): Json<UserSettings>,
30) -> Result<Response, AuthError> {
31    let mut lock = conn.writer().lock_owned().await;
32    let mut tx = dim_database::write_tx(&mut lock).await.map_err(DatabaseError::from)?;
33    let update_user = UpdateableUser {
34        prefs: Some(new_settings.clone()),
35    };
36
37    update_user.update(&mut tx, user.id).await?;
38
39    tx.commit().await.map_err(DatabaseError::from)?;
40    drop(lock);
41
42    Ok(axum::response::Json(&new_settings).into_response())
43}

45fn get_global_settings() -> GlobalSettings {
46    let mut global_settings: GlobalSettings = settings::get_global_settings();
47    let git_tag = String::from(env!("GIT_TAG")).to_owned();
48    let mut git_sha = String::from(env!("GIT_SHA_256")).to_owned();
49    git_sha.truncate(8);
50    let version = git_tag + " " + git_sha.as_str();
51    global_settings.version = version;
52    global_settings
53}
