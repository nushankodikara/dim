22pub async fn banners(
23    Extension(user): Extension<User>,
24    State(AppState { conn, .. }): State<AppState>,
25) -> Result<Response, AuthError> {
26    let mut tx = conn.read().begin().await.map_err(DatabaseError::from)?;
27    let mut banners = Vec::new();
28    for media in Media::get_random_with(&mut tx, 10).await? {
29        if let Ok(x) = match media.media_type {
30            MediaType::Tv => banner_for_show(&mut tx, &user, &media).await,
31            MediaType::Movie => banner_for_movie(&mut tx, &user, &media).await,
32            _ => unreachable!(),
33        } {
34            banners.push(x);
35        }
36    }
37
38    Ok(Json(&banners.iter().take(3).collect::<Vec<_>>()).into_response())
39}

41async fn banner_for_movie(
42    conn: &mut dim_database::Transaction<'_>,
43    user: &User,
44    media: &Media,
45) -> Result<Value, AuthError> {
46    let progress = Progress::get_for_media_user(&mut *conn, user.id, media.id)
47        .await
48        .map(|x| x.delta)
49        .unwrap_or(0);
50
51    let mediafiles = MediaFile::get_of_media(&mut *conn, media.id).await?;
52    let media_duration = MediaFile::get_largest_duration(&mut *conn, media.id).await?;
53
54    let genres = Genre::get_by_media(&mut *conn, media.id)
55        .await
56        .map(|x| x.into_iter().map(|x| x.name).collect::<Vec<_>>())
57        .unwrap_or_default();
58
59    let caption = if progress > 0 {
60        "CONTINUE WATCHING"
61    } else {
62        "WATCH SOMETHING FRESH"
63    };
64
65    Ok(json!({
66        "id": media.id,
67        "title": media.name,
68        "year": media.year,
69        "synopsis": media.description,
70        "backdrop": media.backdrop_path,
71        "duration": media_duration,
72        "genres": genres,
73        "delta": progress,
74        "banner_caption": caption,
75        "versions": mediafiles.iter().map(|x| json!({
76            "id": x.id,
77            "file": x.target_file,
78            "display_name": format!("{} - {} - {} - Library {}",
79                                    x.codec.as_ref().unwrap_or(&"Unknown VC".to_string()),
80                                    x.audio.as_ref().unwrap_or(&"Unknwon AC".to_string()),
81                                    x.original_resolution.as_ref().unwrap_or(&"Unknown res".to_string()),
82                                    x.library_id)
83        })).collect::<Vec<_>>(),
84    }))
85}

87async fn banner_for_show(
88    conn: &mut dim_database::Transaction<'_>,
89    user: &User,
90    media: &Media,
91) -> Result<Value, AuthError> {
92    let episode = if let Ok(Some(ep)) =
93        Episode::get_last_watched_episode(&mut *conn, media.id, user.id).await
94    {
95        let (delta, duration) = Progress::get_progress_for_media(&mut *conn, ep.id, user.id)
96            .await
97            .unwrap_or((0, 1));
98
99        if (delta as f64 / duration as f64) > 0.90 {
100            ep.get_next_episode(&mut *conn).await.unwrap_or(ep)
101        } else {
102            ep
103        }
104    } else {
105        Episode::get_first_for_show(&mut *conn, media.id).await?
106    };
107
108    let genres = Genre::get_by_media(&mut *conn, media.id)
109        .await
110        .unwrap_or_default()
111        .into_iter()
112        .map(|x| x.name)
113        .collect::<Vec<_>>();
114
115    let progress = Progress::get_for_media_user(&mut *conn, user.id, episode.id)
116        .await
117        .map(|x| x.delta)
118        .unwrap_or(0);
119
120    let duration = MediaFile::get_largest_duration(&mut *conn, episode.id)
121        .await
122        .unwrap_or(0);
123
124    let mediafiles = MediaFile::get_of_media(&mut *conn, episode.id).await?;
125
126    let caption = if progress > 0 {
127        "CONTINUE WATCHING"
128    } else {
129        "WATCH SOMETHING FRESH"
130    };
131
132    Ok(json!({
133        "id": episode.id,
134        "title": media.name,
135        "year": media.year,
136        "synopsis": media.description,
137        "backdrop": media.backdrop_path,
138        "duration": duration,
139        "genres": genres,
140        "delta": progress,
141        "banner_caption": caption,
142        "episode": episode.episode,
143        "season": episode.get_season_number(&mut *conn).await.unwrap_or(0),
144        "versions": mediafiles.iter().map(|x| json!({
145            "id": x.id,
146            "file": x.target_file,
147            "display_name": format!("{} - {} - {} - Library {}",
148                                    x.codec.as_ref().unwrap_or(&"Unknown VC".to_string()),
149                                    x.audio.as_ref().unwrap_or(&"Unknwon AC".to_string()),
150                                    x.original_resolution.as_ref().unwrap_or(&"Unknown res".to_string()),
151                                    x.library_id)
152        })).collect::<Vec<_>>(),
153    }))
154}

156pub async fn dashboard(
157    Extension(user): Extension<User>,
158    State(AppState { conn, .. }): State<AppState>,
159) -> Result<Response, AuthError> {
160    let mut tx = conn.read().begin().await.map_err(DatabaseError::from)?;
161    let mut top_rated = Vec::new();
162    for media in Media::get_top_rated(&mut tx, 10).await? {
163        let item = match sqlx::query!(
164            "SELECT _tblmedia.name, assets.local_path FROM _tblmedia LEFT JOIN assets ON assets.id = _tblmedia.poster
165            WHERE _tblmedia.id = ?",
166            media
167        ).fetch_one(&mut tx).await {
168            Ok(x) => x,
169            Err(_) => continue,
170        };
171
172        top_rated.push(json!({
173            "id": media,
174            "poster_path": item.local_path,
175            "name": item.name
176        }));
177    }
178
179    let mut recently_added = Vec::new();
180    for media in Media::get_recently_added(&mut tx, 10).await? {
181        let item = match sqlx::query!(
182            "SELECT _tblmedia.name, assets.local_path FROM _tblmedia LEFT JOIN assets ON assets.id = _tblmedia.poster
183            WHERE _tblmedia.id = ?",
184            media
185        ).fetch_one(&mut tx).await {
186            Ok(x) => x,
187            Err(_) => continue,
188        };
189
190        recently_added.push(json!({
191            "id": media,
192            "poster_path": item.local_path,
193            "name": item.name
194        }));
195    }
196
197    let mut continue_watching = Vec::new();
198    for media in Progress::get_continue_watching(&mut tx, user.id, 10).await? {
199        let item = match sqlx::query!(
200            "SELECT _tblmedia.name, assets.local_path FROM _tblmedia LEFT JOIN assets ON assets.id = _tblmedia.poster
201            WHERE _tblmedia.id = ?",
202            media
203        ).fetch_one(&mut tx).await {
204            Ok(x) => x,
205            Err(_) => continue,
206        };
207
208        continue_watching.push(json!({
209            "id": media,
210            "poster_path": item.local_path,
211            "name": item.name
212        }));
213    }
214
215    let continue_watching = if !continue_watching.is_empty() {
216        Some(json!({
217            "CONTINUE WATCHING": continue_watching,
218        }))
219    } else {
220        None
221    };
222
223    Ok(Json(&json!({
224        ..?continue_watching,
225        "TOP RATED": top_rated,
226        "FRESHLY ADDED": recently_added,
227    })).into_response())
