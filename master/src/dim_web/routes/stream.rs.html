54/// Method mapped to `GET /api/v1/stream/<id>/manifest?<gid>` returns or creates a virtual
55/// manifest.
56pub async fn return_virtual_manifest(
57    State(AppState { conn, state, stream_tracking, .. }): State<AppState>,
58    Path(id): Path<i64>,
59    Query(params): Query<VirtualManifestParams>,
60    Extension(user): Extension<User>,
61) -> Result<impl IntoResponse, errors::StreamingErrors> {
62    let gid = params.gid.and_then(|x| Uuid::parse_str(x.as_str()).ok());
63    if let Some(gid) = gid {
64        return Ok(Json(&json!({
65            "tracks": stream_tracking.get_for_gid(&gid).await,
66            "gid": gid.as_hyphenated().to_string(),
67        })).into_response());
68    }
69
70    let mut tx = conn.read().begin().await?;
71    let user_prefs = user.prefs;
72
73    let gid = uuid::Uuid::new_v4();
74
75    let media = MediaFile::get_one(&mut tx, id)
76        .await
77        .map_err(|e| errors::StreamingErrors::NoMediaFileFound(e.to_string()))?;
78
79    let target_file = media.target_file.clone();
80
81    // FIXME: When `fs::try_exists` gets stabilized we should use that as it will allow us to
82    // detect if the user lacks permissions to access the file, etc.
83    if !path::Path::new(&target_file).exists() {
84        return Err(errors::StreamingErrors::FileDoesNotExist);
85    }
86
87    let info = FFProbeCtx::new(dim_core::streaming::FFPROBE_BIN.as_ref())
88        .get_meta(target_file)
89        .await
90        .map_err(|_| errors::StreamingErrors::FFProbeCtxFailed)?;
91
92    let mut ms = info
93        .get_ms()
94        .ok_or(errors::StreamingErrors::FileIsCorrupt)?
95        .to_string();
96
97    ms.truncate(4);
98
99    let should_stream_default =
100        try_create_dstream(&info, &media, &stream_tracking, &gid, &state, &user_prefs).await?;
101
102    create_video(
103        &info,
104        &media,
105        &stream_tracking,
106        &gid,
107        &state,
108        &user_prefs,
109        should_stream_default,
110    )
111    .await?;
112    create_audio(&info, &media, &stream_tracking, &gid, &state).await?;
113    create_subtitles(&info, &media, &stream_tracking, &gid, &state, params.force_ass).await?;
114
115    stream_tracking.generate_sids(&gid).await;
116
117    Ok(Json(&json!({
118        "tracks": stream_tracking.get_for_gid(&gid).await,
119        "gid": gid.as_hyphenated().to_string(),
120    })).into_response())
121}

123pub async fn try_create_dstream(
124    info: &FFPStream,
125    media: &MediaFile,
126    stream_tracking: &StreamTracking,
127    gid: &Uuid,
128    state: &StateManager,
129    prefs: &UserSettings,
130) -> Result<bool, errors::StreamingErrors> {
131    let video_stream = info
132        .get_primary("video")
133        .cloned()
134        .ok_or(errors::StreamingErrors::FileIsCorrupt)?;
135
136    let ctx = ProfileContext {
137        file: media.target_file.clone(),
138        input_ctx: video_stream.clone().into(),
139        output_ctx: OutputCtx {
140            codec: "h264".into(),
141            start_num: 0,
142            target_gop: 10,
143            ..Default::default()
144        },
145        ..Default::default()
146    };
147
148    let dp_profile_chain =
149        get_profile_for_with_type(StreamType::Video, ProfileType::Transmux, &ctx);
150
151    // Should secondary (transcoded) streams default.
152    let should_stream_default = dp_profile_chain.is_empty()
153        || !matches!(prefs.default_video_quality, DefaultVideoQuality::DirectPlay);
154
155    if !dp_profile_chain.is_empty() {
156        let video = state.create(dp_profile_chain, ctx).await?;
157
158        // FIXME: Stop hardcoding a fps of 24
159        let video_avc = video_stream
160            .level
161            .and_then(|x| level_to_tag(x))
162            .unwrap_or(get_avc1_tag(
163                video_stream.width.clone().unwrap_or(1920) as u64,
164                video_stream.height.clone().unwrap_or(1080) as u64,
165                video_stream
166                    .get_bitrate()
167                    .or(info.get_container_bitrate())
168                    .expect("Failed to pick bitrate for video stream"),
169                24,
170            ));
171
172        let bitrate = video_stream
173            .get_bitrate()
174            .or(info.get_container_bitrate())
175            .unwrap_or(10_000_000);
176
177        let label = {
178            let (ident, bitrate_norm) = if bitrate > 1_000_000 {
179                ("MB", bitrate / 1_000_000)
180            } else {
181                ("KB", bitrate / 1_000)
182            };
183
184            format!(
185                "{}p@{}{} (Direct Play)",
186                video_stream.height.clone().unwrap(),
187                bitrate_norm,
188                ident
189            )
190        };
191
192        let chunk_path = format!("{}/data/$Number$.m4s", video.clone());
193        let init_seg = Some(format!("{}/data/init.mp4", video.clone()));
194        let virtual_manifest =
195            VirtualManifest::new(video.clone(), chunk_path, init_seg, ContentType::Video)
196                .set_direct()
197                .set_mime("video/mp4")
198                .set_duration(info.get_duration())
199                .set_codecs(video_avc.to_string())
200                .set_bandwidth(bitrate)
201                .set_args([("height", video_stream.height.clone().unwrap())])
202                .set_is_default(!should_stream_default)
203                .set_target_duration(10)
204                .set_label(label);
205
206        stream_tracking.insert(&gid, virtual_manifest).await;
207    }
208
209    Ok(should_stream_default)
210}

212pub async fn create_video(
213    info: &FFPStream,
214    media: &MediaFile,
215    stream_tracking: &StreamTracking,
216    gid: &Uuid,
217    state: &StateManager,
218    prefs: &UserSettings,
219    mut should_stream_default: bool,
220) -> Result<(), errors::StreamingErrors> {
221    let video_stream = info
222        .get_primary("video")
223        .cloned()
224        .ok_or(errors::StreamingErrors::FileIsCorrupt)?;
225
226    let qualities = get_qualities(
227        video_stream.height.unwrap_or(1080) as u64,
228        video_stream
229            .get_bitrate()
230            .or(info.get_container_bitrate())
231            .unwrap_or(10_000_000),
232    );
233
234    for quality in qualities {
235        let bitrate = video_stream
236            .get_bitrate()
237            .or(Some(quality.bitrate))
238            .unwrap()
239            .min(quality.bitrate);
240
241        let ctx = ProfileContext {
242            file: media.target_file.clone(),
243            input_ctx: video_stream.clone().into(),
244            output_ctx: OutputCtx {
245                codec: "h264".into(),
246                start_num: 0,
247                bitrate: Some(bitrate),
248                height: Some(quality.height as i64),
249                ..Default::default()
250            },
251            ..Default::default()
252        };
253
254        let global_prefs = dim_core::routes::settings::get_global_settings();
255
256        let profile_chain = get_profile_for(StreamType::Video, &ctx);
257        let profile_chain = if !global_prefs.enable_hwaccel {
258            profile_chain
259                .into_iter()
260                .filter(|x| x.profile_type() != ProfileType::HardwareTranscode)
261                .collect::<Vec<_>>()
262        } else {
263            profile_chain
264        };
265
266        debug_assert!(!profile_chain.is_empty());
267
268        let video = state.create(profile_chain, ctx).await?;
269
270        let video_stream_height = video_stream.height.unwrap_or(1080) as u64;
271        let ratio = video_stream_height as f64 / quality.height as f64;
272        let width = video_stream.width.unwrap_or(1920) as f64 / ratio;
273
274        let video_avc = video_stream
275            .level
276            .and_then(|x| level_to_tag(x))
277            .unwrap_or(get_avc1_tag(
278                width as u64,
279                quality.height,
280                quality.bitrate,
281                24,
282            ));
283
284        let label = quality_to_label(quality.bitrate, quality.height, Some(bitrate));
285
286        // TODO: This code will not work correctly if there are similar resolutions with different
287        // brates.
288        let should_be_default = should_stream_default
289            || matches!(prefs.default_video_quality, DefaultVideoQuality::Resolution(res, _) if res == quality.height);
290
291        let chunk_path = format!("{}/data/$Number$.m4s", video.clone());
292        let init_seg = Some(format!("{}/data/init.mp4", video.clone()));
293        let virtual_manifest =
294            VirtualManifest::new(video.clone(), chunk_path, init_seg, ContentType::Video)
295                .set_mime("video/mp4")
296                .set_duration(info.get_duration())
297                .set_codecs(video_avc.to_string())
298                .set_bandwidth(bitrate)
299                .set_args([("height", quality.height)])
300                .set_is_default(should_be_default)
301                .set_label(label);
302
303        stream_tracking.insert(&gid, virtual_manifest).await;
304        // we wan to default only the first stream.
305        if should_be_default {
306            should_stream_default = false;
307        }
308    }
309    Ok(())
310}

312pub async fn create_audio(
313    info: &FFPStream,
314    media: &MediaFile,
315    stream_tracking: &StreamTracking,
316    gid: &Uuid,
317    state: &StateManager,
318) -> Result<(), errors::StreamingErrors> {
319    let audio_streams = info.find_by_type("audio");
320
321    for stream in audio_streams {
322        let is_default = info.get_primary("audio") == Some(stream);
323        let bitrate = stream
324            .bit_rate
325            .as_ref()
326            .and_then(|x| x.parse::<u64>().ok())
327            .unwrap_or(120_000);
328
329        let ctx = ProfileContext {
330            file: media.target_file.clone(),
331            input_ctx: stream.clone().into(),
332            output_ctx: OutputCtx {
333                codec: "aac".into(),
334                start_num: 0,
335                bitrate: Some(bitrate),
336                ..Default::default()
337            },
338            ..Default::default()
339        };
340
341        let profile = get_profile_for(StreamType::Audio, &ctx);
342        let audio = state.create(profile, ctx).await?;
343
344        let audio_lang = stream
345            .get_language()
346            .as_deref()
347            .and_then(dim_core::utils::lang_from_iso639)
348            .unwrap_or("Unknown");
349
350        let audio_codec = dim_core::utils::codec_pretty(stream.get_codec());
351        let audio_ch = dim_core::utils::channels_pretty(stream.channels.unwrap_or(2));
352
353        let label = format!("{} ({} {})", audio_lang, audio_codec, audio_ch);
354
355        let chunk_path = format!("{}/data/$Number$.m4s", audio.clone());
356        let init_seg = Some(format!("{}/data/init.mp4", audio.clone()));
357        let virtual_manifest =
358            VirtualManifest::new(audio.clone(), chunk_path, init_seg, ContentType::Audio)
359                .set_mime("audio/mp4")
360                .set_codecs("mp4a.40.2")
361                .set_bandwidth(bitrate)
362                .set_is_default(is_default)
363                .set_label(label)
364                .set_lang(stream.get_language());
365
366        stream_tracking.insert(&gid, virtual_manifest).await;
367    }
368
369    Ok(())
370}

372pub async fn create_subtitles(
373    info: &FFPStream,
374    media: &MediaFile,
375    stream_tracking: &StreamTracking,
376    gid: &Uuid,
377    state: &StateManager,
378    force_ass: bool,
379) -> Result<(), errors::StreamingErrors> {
380    let subtitles = info.find_by_type("subtitle");
381
382    for stream in subtitles {
383        let is_default = info.get_primary("subtitle") == Some(stream);
384        let is_ssa = ["ssa", "ass"].contains(&stream.codec_name.as_str()) && force_ass;
385
386        if !["subrip", "ass", "ssa", "srt", "webvtt", "vtt"].contains(&stream.codec_name.as_str()) {
387            // FIXME: hdmv_pgs_subtitle are not supported yet.
388            continue;
389        }
390
391        let (mime, codec, output_codec) = if is_ssa {
392            ("text/ass", "ass", "ass")
393        } else {
394            ("text/vtt", "vtt", "webvtt")
395        };
396
397        let ctx = ProfileContext {
398            file: media.target_file.clone(),
399            input_ctx: stream.clone().into(),
400            output_ctx: OutputCtx {
401                codec: output_codec.into(),
402                outdir: "-".into(),
403                ..Default::default()
404            },
405            ..Default::default()
406        };
407
408        let lang = stream
409            .get_language()
410            .as_deref()
411            .and_then(dim_core::utils::lang_from_iso639)
412            .unwrap_or("Unknown")
413            .to_string();
414
415        let title = stream.get_title().unwrap_or(lang.clone());
416
417        let profile_chain = get_profile_for(StreamType::Subtitle, &ctx);
418        let subtitle = state.create(profile_chain, ctx).await?;
419
420        let chunk_path = if is_ssa {
421            format!("{}/data/stream.ass", subtitle.clone())
422        } else {
423            format!("{}/data/stream.vtt", subtitle.clone())
424        };
425
426        let virtual_manifest =
427            VirtualManifest::new(subtitle.clone(), chunk_path, None, ContentType::Subtitle)
428                .set_mime(mime)
429                .set_codecs(codec)
430                .set_bandwidth(1024)
431                .set_is_default(is_default)
432                .set_label(stream.get_title().unwrap_or(lang.clone()))
433                .set_lang(stream.get_language());
434
435        let title = title.replace("&", "and"); // dash.js seems to note like when there are `&` within titles.
436        let virtual_manifest = virtual_manifest.set_args([("title".to_string(), title)]);
437
438        stream_tracking.insert(&gid, virtual_manifest).await;
439    }
440
441    Ok(())
442}

451/// Method mapped to `/api/v1/stream/<gid>/manifest.mpd` compiles a virtual manifest into a
452/// mpeg-dash manifest.
453///
454/// # Query args
455/// * `start_num` - first chunk number
456/// * `should_kill` - indicates whether we should clean old streams up while compiling the
457/// manifest.
458/// * `includes` - ids of streams to include, comma separated.
459pub async fn return_manifest(
460    State(AppState { state, stream_tracking, .. }): State<AppState>,
461    Path(gid): Path<String>,
462    Query(params): Query<ManifestParams>,
463) -> Result<impl IntoResponse, errors::StreamingErrors> {
464    let gid = match Uuid::parse_str(gid.as_str()) {
465        Ok(x) => x,
466        Err(_) => return Err(errors::StreamingErrors::GidParseError),
467    };
468    if params.should_kill.unwrap_or(true) {
469        let ids = stream_tracking
470            .get_for_gid(&gid)
471            .await
472            .into_iter()
473            .filter(|x| !matches!(x.content_type, ContentType::Video | ContentType::Audio))
474            .map(|x| x.id)
475            .collect::<Vec<_>>();
476        stream_tracking.kill(&state, &gid, ids, true).await;
477    }
478
479    let manifest = if let Some(includes) = params.includes {
480        let includes = includes
481            .split(",")
482            .map(ToString::to_string)
483            .collect::<Vec<_>>();
484
485        stream_tracking
486            .compile_only(&gid, params.start_num.unwrap_or(0), includes)
487            .await
488            .unwrap()
489    } else {
490        stream_tracking
491            .compile(&gid, params.start_num.unwrap_or(0))
492            .await
493            .unwrap()
494    };
495
496    Ok(([(header::CONTENT_TYPE, "application/dash+xml")], manifest))
497}

499/// Repeatedly invoke a nightfall routine until a timeout occurs waiting for a chunk to be "ready".
500///
501/// `tick_dur` will the the duration amount that gets passed into `std::thread::sleep` and it will
502/// block for AT MOST `tick_limit` ticks. When a the tick limit has been hit `None` is returned
503/// otherwise `Some(Result<T, NightfallError>)` is returned.
504///
505async fn timeout_segment<F, T>(
506    f: impl Fn() -> F,
507    tick_dur: Duration,
508    tick_limit: usize,
509) -> Result<T, NightfallError>
510where
511    F: Future<Output = Result<T, NightfallError>>,
512{
513    let mut ticks = 0usize;
514
515    loop {
516        if ticks >= tick_limit {
517            return Err(NightfallError::ChunkNotDone.into());
518        }
519
520        let result = f().await;
521
522        if let Err(NightfallError::ChunkNotDone) = result {
523            ticks += 1;
524            tokio::time::sleep(tick_dur).await;
525        } else {
526            break result;
527        }
528    }
529}

536/// Method mapped to `/api/v1/stream/<id>/data/init.mp4` returns the init chunk of the stream `id`.

538/// # Query args
539/// * `start_num` - first chunk index

542    Path(id): Path<String>,
543    Query(params): Query<InitParams>

546        || state.chunk_init_request(id.clone(), params.start_num.unwrap_or(0)),
547        Duration::from_millis(100),

552    Ok(reply_with_file(path, ("Content-Type", "video/mp4")).await)

555/// Method mapped to `/api/v1/stream/<id>/data/<chunk..>` returns a chunk for stream `id`.
556pub async fn get_chunk(
557    State(AppState { state, .. }): State<AppState>,
558    Path((id, chunk)): Path<(String, PathBuf)>,
559) -> Result<impl IntoResponse, errors::StreamingErrors> {
560    let extension = chunk
561        .extension()
562        .ok_or(errors::StreamingErrors::InvalidRequest)?
563        .to_string_lossy()
564        .into_owned();
565
566    // Chunks will always be m4s or mp4
567    if extension.as_str() != "m4s" {
568        return Err(errors::StreamingErrors::InvalidRequest);
569    }
570
571    // Parse the chunk filename into a u64, we unwrap_or because sometimes it can be a init chunk,
572    // if its a init chunk we assume a chunk index of 0 because we are fetching the first few
573    // chunks.
574    let chunk_num = chunk
575        .file_stem()
576        .ok_or(errors::StreamingErrors::InvalidRequest)?
577        .to_string_lossy()
578        .into_owned()
579        .parse::<u32>()
580        .unwrap_or(0);
581
582    let path: String = timeout_segment(
583        || state.chunk_request(id.clone(), chunk_num),
584        Duration::from_millis(100),
585        100,
586    )
587    .await?;
588
589    Ok(reply_with_file(path, ("Content-Type", "video/mp4")).await)
590}

611/// Method mapped to `/api/v1/stream/<id>/data/stream.ass` attempts to transcode the underlying
612/// stream to ASS.
613///
614/// # Arguments
615/// * `id` - id of the underlying stream (Must be a subtitle stream of non-bitmap format).
616pub async fn get_subtitle_ass(
617    State(AppState { state, .. }): State<AppState>,
618    Path(id): Path<String>,
619) -> Result<impl IntoResponse, errors::StreamingErrors> {
620    let path: String = timeout_segment(
621        || async {
622            if state.has_started(id.clone()).await.unwrap_or(false) {
623                if state.is_done(id.clone()).await.unwrap_or(false) {
624                    return state.get_sub(id.clone(), "stream".into()).await;
625                }
626            } else {
627                let _ = state.start(id.clone()).await;
628            }
629
630            Err(NightfallError::ChunkNotDone)
631        },
632        Duration::from_millis(100),
633        200,
634    )
635    .await?;
636
637    Ok(reply_with_file(path, ("Content-Type", "text/ass")).await)
638}

640/// Method mapped to `/api/v1/stream/<gid>/state/should_hard_seek/<chunk_num>` returns whether the
641/// client should hard seek in order to play the video at `chunk_num`. This is really only useful
642/// on web platforms.
643pub async fn should_client_hard_seek(
644    State(AppState { state, stream_tracking, .. }): State<AppState>,

646) -> Result<impl IntoResponse, errors::StreamingErrors> {

650    };
651    let ids = stream_tracking.get_for_gid(&gid).await;
652
653    let mut should_client_hard_seek = false;

657    }
658

660        "should_client_seek": should_client_hard_seek,
661    })).into_response())

668    Path(gid): Path<String>,
669) -> Result<impl IntoResponse, errors::StreamingErrors> {
670    let gid = match Uuid::parse_str(gid.as_str()) {
671        Ok(x) => x,
672        Err(_) => return Err(errors::StreamingErrors::GidParseError),
673    };
674    Ok(Json(&json!({
675    "errors": stream::iter(stream_tracking
676        .get_for_gid(&gid)
677        .await)
678        .filter_map(|x| async { state.get_stderr(x.id).await.ok() })
679        .collect::<Vec<_>>().await,
680    })).into_response())
681}

683/// Method mapped to `/api/v1/stream/<gid>/state/kill` will kill all streams for `gid`.
684pub async fn kill_session(
685    State(AppState { state, stream_tracking, .. }): State<AppState>,
686    Path(gid): Path<String>,
687) -> Result<impl IntoResponse, errors::StreamingErrors> {
688    let gid = match Uuid::parse_str(gid.as_str()) {
689        Ok(x) => x,
690        Err(_) => return Err(errors::StreamingErrors::GidParseError),
691    };
692    for manifest in stream_tracking.get_for_gid(&gid).await {
693        let _ = state.die(manifest.id).await;
694    }
695
696    Ok(StatusCode::NO_CONTENT)
697}
